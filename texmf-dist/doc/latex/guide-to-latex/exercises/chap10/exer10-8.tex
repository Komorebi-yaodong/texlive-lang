\documentclass{article}
\newcounter{comment}
\newsavebox{\comname}
\newenvironment{comment}[1]
  {\noindent\stepcounter{comment}\slshape Kommentar \arabic{comment}\nopagebreak
   \begin{quote}\sloppypar\small\sbox{\comname}{#1}\itshape}
  {\nopagebreak\hspace*{\fill}\usebox{\comname}\end{quote}}
\begin{document}
\begin{comment}{Patrick W. Daly}
   In this form, every comment is assigned a sequential numbr after the
   word Comment. The comment text appears as in a \texttt{quote} environment,
   while the name of the commentor is entered as the envrironment argument
   and is placed at the right of the last line. In addition no page break
   can occur either between the heading `Comment n' and the comment text
   and the commenter's name.
\end{comment}
\end{document}
